> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking. **Do NOT run `git commit` or `git add` — the user has explicit rules against unauthorized commits.**

**Goal:** Auto-classify each clinic account as tier A/B/C based on delivered revenue (20/30/50 percentile split, per-rep scope for reps and global scope for admins), with badges on the accounts list + detail page and a breakdown section on rep-performance.

**Architecture:** Tiers are computed live on every request by a pure `assignTiers()` helper and attached to `IAccountWithMetrics.tier`. No schema changes. Both `getAccountsWithMetrics` and a new `buildTierCounts` helper in the rep-performance action call the same utility. UI additions are a shared `AccountTierBadge` component plus a new `TierBreakdown` section on rep-performance.

**Tech Stack:** Next.js 16 App Router server actions, Supabase admin client, Redux Toolkit, TypeScript, Tailwind CSS 4.

## Task 3: Create `assignTiers()` helper

**Files:**
- Create: `utils/helpers/tiers.ts`

- [ ] **Step 1: Create the file**

```typescript
import type { AccountTier } from "@/utils/interfaces/accounts";

/**
 * Minimal shape an input account must satisfy: an id and a delivered_revenue.
 * The function is generic so callers can pass either `IAccountWithMetrics`
 * (accounts page) or a slim `{ id, delivered_revenue }` shape (rep-performance
 * tier counts) and still get back the same shape with a `tier` field.
 */
export interface TierInput {
  id: string;
  delivered_revenue: number;
}

export function assignTiers<T extends TierInput>(accounts: T[]): Array<T & { tier: AccountTier }> {
  if (accounts.length === 0) return [];

  // Partition: zero-revenue accounts are forced to C regardless of rank.
  const hasRevenue = accounts.filter((a) => (a.delivered_revenue ?? 0) > 0);
  const zeroRevenue = accounts.filter((a) => (a.delivered_revenue ?? 0) <= 0);

  // Sort revenue-bearing accounts descending, assign percentile tiers.
  const sorted = [...hasRevenue].sort(
    (a, b) => (b.delivered_revenue ?? 0) - (a.delivered_revenue ?? 0),
  );

  const aCount = Math.ceil(sorted.length * 0.20);
  const bCount = Math.ceil(sorted.length * 0.30);

  const tierById = new Map<string, AccountTier>();
  sorted.forEach((a, i) => {
    if (i < aCount) tierById.set(a.id, "A");
    else if (i < aCount + bCount) tierById.set(a.id, "B");
    else tierById.set(a.id, "C");
  });
  for (const a of zeroRevenue) tierById.set(a.id, "C");

  // Return input in original order with tier attached.
  return accounts.map((a) => ({ ...a, tier: tierById.get(a.id) ?? "C" }));
}
```

```typescript
async function buildTierCounts(
  repId: string | null,
  adminClient: ReturnType<typeof createAdminClient>,
): Promise<{ A: number; B: number; C: number }> {
  // Resolve facility scope for the viewer.
  let facilityIds: string[] | null = null;
  if (repId) {
    const { data: facs } = await adminClient
      .from("facilities")
      .select("id")
      .eq("assigned_rep", repId);
    facilityIds = (facs ?? []).map((f: any) => f.id as string);
    if (facilityIds.length === 0) return { A: 0, B: 0, C: 0 };
  } else {
    const { data: facs } = await adminClient
      .from("facilities")
      .select("id")
      .eq("facility_type", "clinic");
    facilityIds = (facs ?? []).map((f: any) => f.id as string);
    if (facilityIds.length === 0) return { A: 0, B: 0, C: 0 };
  }

  // Trailing-3-month delivered orders for these facilities.
  const now = new Date();
  const start = new Date(now.getFullYear(), now.getMonth() - 3, now.getDate()).toISOString();

  const { data: deliveredOrders } = await adminClient
    .from("orders")
    .select("id, facility_id")
    .in("facility_id", facilityIds)
    .eq("delivery_status", "delivered")
    .gte("delivered_at", start);

  const deliveredIds = (deliveredOrders ?? []).map((o: any) => o.id as string);
  const revenueByOrder: Record<string, number> = {};
  if (deliveredIds.length > 0) {
    const { data: items } = await adminClient
      .from("order_items")
      .select("order_id, total_amount")
      .in("order_id", deliveredIds);
    for (const it of items ?? []) {
      revenueByOrder[it.order_id] =
        (revenueByOrder[it.order_id] ?? 0) + Number(it.total_amount ?? 0);
    }
  }

  // Sum delivered revenue per facility.
  const revenueByFacility: Record<string, number> = {};
  for (const o of deliveredOrders ?? []) {
    const fid = o.facility_id as string;
    revenueByFacility[fid] =
      (revenueByFacility[fid] ?? 0) + (revenueByOrder[o.id] ?? 0);
  }

  // Build tier inputs for every facility in scope (including zero-revenue ones).
  const tierInputs = facilityIds.map((id) => ({
    id,
    delivered_revenue: revenueByFacility[id] ?? 0,
  }));

  const tiered = assignTiers(tierInputs);
  let A = 0, B = 0, C = 0;
  for (const t of tiered) {
    if (t.tier === "A") A += 1;
    else if (t.tier === "B") B += 1;
    else C += 1;
  }
  return { A, B, C };
}
```
